import { Canvas, Meta, Story } from "@storybook/addon-docs"
import { NumberInput, NumberInputField } from "@chakra-ui/react"
import { useState } from 'react'
import { argTypes } from "./argTypes"

<Meta title="atoms/Input" component={NumberInput} />

export const Template = (args) => {
  const [value, setValue] = useState("")
  const changeHandler = (value) => {
    setValue(value)
  }
  return (
    <NumberInput {...args}>
      <NumberInputField placeholder={args.placeholder} value={value} onChange={e => setValue(e.target.value)} />
    </NumberInput>
  )
}

<Canvas>
  <Story
    name="NumberInput"
    argTypes={argTypes}
    args={{
      placeholder: "100",
    }}
  >
    {Template.bind({})}
  </Story>
</Canvas>
